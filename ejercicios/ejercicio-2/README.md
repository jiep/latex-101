% Añadir paquete para fórmulas amsmath

\title{Taller LaTeX -- Ejercicio 2}
\author{Tu nombre}
\date{26 de junio de 2019}

\begin{document}

\maketitle

\tableofcontents

% Añadir en cada sección la fórmula correspondiente

\section{Introduction}

\lipsum[2]

\end{document}
